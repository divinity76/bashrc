```bash
mv ~/.bashrc ~/.bashrc_old ;
mv ~/.bash_aliases ~/.bash_aliases_old ;

wget https://github.com/divinity76/bashrc/raw/master/.bash_aliases -O ~/.bash_aliases ;
source ~/.bashrc

```
